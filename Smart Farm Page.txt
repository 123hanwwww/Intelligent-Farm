        function updateData() {
            
            const tempElement = document.querySelector('.data-grid .data-item:nth-child(1) .data-value-small');
            const currentTemp = parseInt(tempElement.textContent);
            const newTemp = currentTemp + (Math.random() * 2 - 1);
            const trendElement = tempElement.querySelector('.trend');
            const change = (newTemp - currentTemp).toFixed(1);
            
            if (change > 0) {
                trendElement.textContent = `+${change}°`;
                trendElement.className = 'trend trend-up';
            } else {
                trendElement.textContent = `${change}°`;
                trendElement.className = 'trend trend-down';
            }
            
            tempElement.innerHTML = `${newTemp.toFixed(1)}°C ${trendElement.outerHTML}`;
            
            
            const healthScoreElement = document.querySelector('.data-value');
            const currentScore = parseInt(healthScoreElement.textContent);
            const newScore = Math.max(0, Math.min(100, currentScore + (Math.random() * 4 - 2)));
            healthScoreElement.textContent = `${Math.round(newScore)}/100`;
            
            
            const healthDot = document.querySelector('.card:nth-child(1) .status-dot');
            if (newScore > 80) {
                healthDot.className = 'status-dot healthy';
            } else if (newScore > 60) {
                healthDot.className = 'status-dot warning';
            } else {
                healthDot.className = 'status-dot danger';
            }
            
                        const cameraProgress = document.querySelector('.hardware-item:nth-child(1) .progress');
            const currentCameraProgress = parseInt(cameraProgress.style.width);
            const newCameraProgress = Math.max(50, Math.min(80, currentCameraProgress + (Math.random() * 10 - 5)));
            cameraProgress.style.width = `${newCameraProgress}%`;
            
            if (newCameraProgress > 70) {
                cameraProgress.className = 'progress progress-healthy';
            } else if (newCameraProgress > 50) {
                cameraProgress.className = 'progress progress-warning';
            } else {
                cameraProgress.className = 'progress progress-danger';
            }
        }
        
       
        setInterval(updateData, 10000);
        
        
        document.querySelectorAll('button')[0].addEventListener('click', updateData);
        document.querySelectorAll('button')[1].addEventListener('click', function() {
