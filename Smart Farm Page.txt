    <script>
        function updateData() {
            const healthValue = document.querySelector('.card .value');
            const currentScore = parseInt(healthValue.textContent);
            const newScore = Math.max(0, Math.min(100, currentScore + (Math.random() * 4 - 2)));
            healthValue.textContent = `${Math.round(newScore)}/100`;
            
            const healthDot = document.querySelector('.card:nth-child(1) .dot');
            if (newScore > 80) {
                healthDot.className = 'dot healthy';
            } else if (newScore > 60) {
                healthDot.className = 'dot warning';
            } else {
                healthDot.className = 'dot danger';
            }
            
            const tempElement = document.querySelector('.card:nth-child(2) .data-item:nth-child(1)');
            const currentTemp = parseInt(tempElement.textContent);
            const newTemp = currentTemp + (Math.random() * 2 - 1);
            tempElement.textContent = `Temp: ${newTemp.toFixed(1)}Â°C`;
        }
        
        setInterval(updateData, 10000);
        
        document.querySelectorAll('button')[0].addEventListener('click', updateData);
        document.querySelectorAll('button')[1].addEventListener('click', function() {
            alert('Generating report...');

            alert('Opening settings');
        });
